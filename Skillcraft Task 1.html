# Distribution of ages in a population.

df = pd.read_csv('Age_Group_Population.csv')

plt.bar(df['Age Group'], df['Population (Mn)'], color=['yellow', 'blue', 'pink'], edgecolor='black')

plt.title("India's Population Distribution by Age in 2022")
plt.xlabel("Age Group")
plt.ylabel("Population (in Millions)")

for i, value in enumerate(df['Population (Mn)']):
    plt.text(i, value + 10, f"{df['Percentage'][i]}%", ha='center', fontsize=10)

plt.show()
